// Array sum program
static void Main(string[] args)
        {
            Console.Write("Enter a array: ");
            int a = int.Parse(Console.ReadLine());
            int[] b = new int[a];
            int i = 0;
            int sum = 0;

            while (i <a)
            {
                Console.WriteLine("Ener a number :");
                b[i] = int.Parse(Console.ReadLine());
                sum = sum + b[i];
                i++;
            }
            Console.WriteLine("sum of array :"+ sum);
            Console.ReadLine();


        }
    }
}
